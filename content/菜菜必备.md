# 字典序

## 字母
### 比较规则回顾：

1. **从左到右逐个字符比较**
    
2. **比较字符的ASCII值**
    
3. **遇到不同字符时，ASCII值小的排在前面**
    
4. **如果一个字符串是另一个的前缀，较短者排在前面**

## 数字

### 基本规则：

- **将数字转换为字符串**
    
- **从左到右逐个字符比较**
    
- **比较字符的ASCII值**
    
- **如果位数不同，较短的字符串在比较时相当于后面补空字符**

## 复合词缩写

缩写为单词首字母的大写
```cpp
#include <iostream>  
#include <string>  
#include <cctype>  
using namespace std;  
  
int main() {  
    string s;  
    getline(cin, s);  // 读取整行输入  
    
    string result = "";  
    bool newWord = true;  // 标记是否是新单词的开始  
  
    for (char c : s) {  
        if (isalpha(c)) {  // 如果是字母  
            if (newWord) {  
                result += toupper(c);  // 转换为大写并添加到结果  
                newWord = false;  
            }  
        } else {  // 如果是非字母字符（空格、标点等），则下一个字母是新单词的开始  
            newWord = true;  
        }  
    }  
    
    cout << result << endl;  
    return 0;  
}
```

## 高精度乘除处理

```cpp

vector<int> mul(const vector<int> &a, int b) {  
    vector<int> res;  
    int carry = 0;  
    for (int x: a) {  
        carry += x * b;  
        res.push_back(carry % 10);  
        carry /= 10;  
    }    while (carry) {  
        res.push_back(carry % 10);  
        carry /= 10;  
    }    return res;  
}  
  
vector<int> div(const vector<int> &a, int b) {  
    vector<int> res;  
    int la = 0;  
    for (int i = a.size() - 1; i >= 0; i--) {  
        la = la * 10 + a[i];  
        res.push_back(la / b);  
        la %= b;  
    }    reverse(res.begin(), res.end());  
    while (res.size() > 1 && res.back() == 0) res.pop_back();  
    return res;  
}
```

## 字符串去重查找

```cpp


int main() {
    int n;                      // 行数（或组数）
    cin >> n;
    // 全局映射：字符串 -> 它出现的所有行号（自动去重且升序）
    map<string, set<int>> a;
    // 读入 n 行数据
    for (int i = 1; i <= n; i++) {
        int num;                // 当前行的字符串个数
        cin >> num;
        set<string> b;          // 用于对当前行内的字符串去重
        for (int j = 0; j < num; j++) {
            string s;
            cin >> s;
            // 如果 s 在当前行第一次出现，则插入去重集合并记录行号
            if (b.insert(s).second) {// .second 表示是否插入成功
                a[s].insert(i); // 将当前行号 i 加入字符串 s 的集合
            }
        }
    }

    int q;                      // 查询次数
    cin >> q;
    while (q--) {
        string ss;
        cin >> ss;

        // 查询字符串 ss 是否出现在任何行中
        if (a.find(ss) != a.end()) {
            // 输出所有出现过的行号，第一个前面不加空格，后续加空格分隔
            auto it = a[ss].begin();
            cout << *it;
            it++;
            for (; it != a[ss].end(); it++) {
                cout << ' ' << *it;
            }
        }
        // 无论是否找到，都输出换行（找不到时仅输出空行）
        cout << '\n';
    }

    return 0;
}
```
