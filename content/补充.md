## 多个子串的最小修改次数
```cpp
string s1 = "awdec";  
string s2 = "Fantasy_Blue";  
int n, k;  
string s;  
int l1 = s1.size(), l2 = s2.size();  
  
void solve() {  
    cin >> n >> k >> s;  
  
    if ((int) s.size() < 17) {  
        cout << "No\n";  
        return;  
    }  
    if (k >= 17) {  
        cout << "Yes\n";  
        return;  
    }  
  
    i64 na = -1, nb = -1;  
    vector<int> w1; // 存储达到最大匹配的起始位置  
    for (int i = 0; i <= n - l2; i++) {  
        i64 cnt = 0;  
        for (int j = 0; j < l2; j++) {  
            if (s[i + j] == s2[j]) cnt++;  
        }  
        if (cnt >= na) {  
            if (cnt > na) w1.clear();  
            na = cnt;  
            w1.push_back(i);  
        }  
    }  
  
    int l = w1[0] + l2;  
    int r = w1.back();  
    // 先搜索 [0, min(r-5, n-5)] 区间  
    for (int i = 0; i <= r - 5; i++) {  
        i64 cnt = 0;  
        for (int j = 0; j < l1; j++) {  
            if (s[i + j] == s1[j]) cnt++;  
        }  
        if (cnt > nb) nb = cnt;  
    }  
    // 再搜索 [l, n-5] 区间  
    for (int i = l; i <= n - 5; i++) {  
        i64 cnt = 0;  
        for (int j = 0; j < l1; j++) {  
            if (s[i + j] == s1[j]) cnt++;  
        }  
        if (cnt > nb) nb = cnt;  
    }  
  
    if ( na + nb + k < 17)cout << "No\n";  
    else cout << "Yes\n";  
}  
```
## 三角形计数（等腰底平行）
```cpp

void solve() {  
    int n;  
    cin >> n;  
    map<int, vector<int>> ax, ay;  
    map<int, int> cx, cy;  
    map<pair<int, int>, int> vis;  
  
    for (int i = 0; i < n; i++) {  
        int x, y;  
        cin >> x >> y;  
        ax[y].push_back(x);  
        ay[x].push_back(y);  
        cx[x]++, cy[y]++;  
        vis[{x, y}] = 1;  
    }  
    i64 ans = 0;  
  
    for (auto &[y, xs]: ax) {  
        int m = xs.size();  
        for (int i = 0; i < m; i++) {  
            for (int j = i + 1; j < m; j++) {  
                int x1 = xs[i], x2 = xs[j];  
                if ((x1 + x2) % 2) continue;  
                int mid = (x1 + x2) / 2;  
                int tp = cx[mid];  
                if (vis.count({mid, y})) tp--;  
                ans += tp;  
            }  
        }  
    }  
  
    for (auto &[x, ys]: ay) {  
        int m = ys.size();  
        for (int i = 0; i < m; i++) {  
            for (int j = i + 1; j < m; j++) {  
                int y1 = ys[i], y2 = ys[j];  
                if ((y1 + y2) % 2) continue;  
                int mid = (y1 + y2) / 2;  
                int tp = cy[mid];  
                if (vis.count({x, mid})) tp--;  
                ans += tp;  
            }  
        }  
    }  
  
    cout << ans << '\n';  
}
```
