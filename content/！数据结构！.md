```cpp
vector<int> fa(mxv), siz(mxv)；
int n, m;

int find(int x) {
    if (fa[x] == x) return x;
    else return fa[x] = find(fa[x]);

}
void solve() {
    cin >> n >> m;
    for (int i = 1; i <= n; ++i) fa[i] = i, siz[i] = 1;
    for (int i = 1; i <= m; ++i) {
        int op, x, y, z;
        cin >> op;
    	if(op == 1){ // 合并操作 
			x = find(x), y = find(y); // 查询各自的代表元素
			if(x == y) continue; // 如果已经在同一个集合，跳过
			if(siz[y] < siz[x]) swap(x, y); // 改变合并顺序
			siz[y] += siz[x]; // 计算新的秩
			fa[x] = y; // 合并 
		}
		else if(op == 2){ // 查询操作的一种变形 
			x = find(x), y = find(y); // 查询各自的代表元素
			if(x == y) cout << "Y" << endl; // 是否在同一个集合
			else       cout << "N" << endl; 
		}else {
            cin >> z;
            z = find(z);
            cout << siz[z] << '\n';
        }
    }
    return ;
}
```

## 带权并查集

```cpp

struct DSU {
    vector<int> fa, dt;
    explicit DSU(int n) : fa(n, -1), dt(n, 0){}

    int find(int x) {
        if (fa[x] < 0) return x;
        int rt = find(fa[x]);
        dt[x] = (dt[x] + dt[fa[x]]) % 2;
        fa[x] = rt;
        return rt;
    }

    int dist(int x, int y) {
        int rx = find(x), ry = find(y);
        if (rx != ry) return inf; 
        return (dt[y] - dt[x] + 2) % 2;
    }

    int merge(int x, int y, int d) {
        int rx = find(x), ry = find(y);
        if (rx == ry) {
            int curd = (dt[y]^dt[x]);
            return curd == d ? rx : -1;
        }
        int newd = (d ^ dt[y] ^ dt[x] );
        if (fa[rx] < fa[ry]) {
            // rx的集合更大,fa值是负数
            fa[rx] += fa[ry];//根节点记录子节点数量
            fa[ry] = rx, dt[ry] = newd;
            return rx;
        } else {
            fa[ry] += fa[rx];
            fa[rx] = ry, dt[rx] = newd;
            return ry;
        }
    }
};

void solve() {
    int n;
    cin >> n;
    vector<tuple<int, int, int> > ed(n);
    vector<int> num;
    for (int i = 0; i < n; i++) {
        int a, b, c;
        cin >> a >> b >> c;
        ed[i] = {a, b, c};
        num.push_back(a);
        num.push_back(b);
    }
    // 离散化
    sort(num.begin(), num.end());
    num.erase(unique(num.begin(), num.end()), num.end());
    int sz = num.size();

    DSU dsu(sz);

    auto id = [&](int x) {
        //查询原始编号x对应的离散化编号
        return lower_bound(num.begin(), num.end(), x) - num.begin();//减去 num.begin() 得到下标（从0开始）
    };

    for (auto [a, b, c]: ed) {
        int x = id(a);
        int y = id(b);
        int d = (c == 1) ? 0 : 1; // 友离0，敌离1
        if (dsu.merge(x, y, d) == -1) {
            cout << "NO" << '\n';
            return;
        }
    }
    cout << "YES" << '\n';
}
```

## ST表（静态数据查最值)
```cpp

void solve() {  
    int n, q;  
    cin >> n >> q;  
    vector<int> a(n + 1);  
    vector<int> log(n + 1);  
    for (int i = 1; i <= n; i++) cin >> a[i];  
  
    // 预处理对数表：O(n)计算所有⌊log₂(i)⌋  
    log[1] = 0;  
    for (int i = 2; i <= n; i++) log[i] = log[i >> 1] + 1;  // i/2的log值加1  
  
    // 确定最大幂次K：2^K <= n 的最大整数K  
    int K = log[n] + 1;  
  
    // ---------- 2. ST表定义 ----------    // 第一维k: 幂次，第二维i: 起点  
    vector mina(K + 1, vector<int>(n + 1));  
    vector maxa(K + 1, vector<int>(n + 1));  
  
    // ---------- 3. 初始化 ----------    // k=0: 区间长度为2^0=1，最小值/最大值就是元素本身  
    for (int i = 1; i <= n; i++) {  
        mina[0][i] = a[i];  
        maxa[0][i] = a[i];  
    }  
  
    // ---------- 4. 构建ST表 ----------    // 核心思想：用两个2^(k-1)长度的区间覆盖2^k长度的区间  
    for (int k = 1; k <= K; k++) {  
        int len = (1 << k);  // 当前区间长度 = 2^k        for (int i = 1; i + len - 1 <= n; i++) {  
            // 从i开始，长度为2^k的区间分成两部分：  
            // 左半: [i, i+2^(k-1)-1]  右半: [i+2^(k-1), i+2^k-1]  
            mina[k][i] = min(mina[k - 1][i], mina[k - 1][i + (1 << (k - 1))]);  
            maxa[k][i] = max(maxa[k - 1][i], maxa[k - 1][i + (1 << (k - 1))]);  
        }  
    }  
    // 从i开始，长度为2^k的区间最小值 = min(左半部分最小值, 右半部分最小值)  
  
    // ---------- 5. 查询处理 ----------    while (q--) {  
        int op, l, r, ans;  
        cin >> op >> l >> r;  
  
        int len = r - l + 1;      // 查询区间长度  
        int k = log[len];         // 找到最大的k使得2^k ≤ len  
  
        // 查询技巧：用两个长度为2^k的区间完全覆盖[l,r]  
        // 两个区间：[l, l+2^k-1] 和 [r-2^k+1, r]        if (op == 1)  
            ans = min(mina[k][l], mina[k][r - (1 << k) + 1]);  
        else  
            ans = max(maxa[k][l], maxa[k][r - (1 << k) + 1]);  
  
        cout << ans << '\n';  
    }  
}  
```

## 树状数组
```cpp
struct BIT {  
    int n;  
    vector<i64> tr;  
    BIT(int size) : n(size), tr(size + 5, 0) {}  
  
    void add(int p, i64 x) {  
        for (; p <= n; p += p & -p) tr[p] += x;  
    }  
  
    i64 ask(int p) {//计算前缀和  
        i64 res = 0;  
        for (; p; p -= p & -p) res += tr[p];//将当前节点贡献加入结果后，跳到前一个不相交的区间。  
        return res;  
    }  
  
    // 区间 [l,r] 和  
    i64 query(int l, int r) {  
        return ask(r) - ask(l - 1);  
    }  
};  
  
void solve() {  
    int n, q;  
    cin >> n >> q;  
    vector<int> a(n + 1);  
    BIT bit(n);  
  
    for (int i = 1; i <= n; i++) {  
        cin >> a[i];  
        bit.add(i, a[i]);  
    }  
  
    while (q--) {  
        int op; cin >> op;  
        if (op == 1) {  
            // 单点修改 a[pos] += x            int pos, x;  
            cin >> pos >> x;  
            bit.add(pos, x);  
        } else {  
            // 区间求和  
            int l, r;  
            cin >> l >> r;  
            cout << bit.query(l, r) << '\n';  
        }  
    }  
}
```

## 线段树

```cpp

// 线段树结构体封装  
struct SegmentTree {  
    vector<int> sumv;  
    vector<bool> lazy;  
    string s;  
  
    // 初始化  
    void init(const string& str) {  
        s = str;  
        int n = s.length();  
        sumv.assign(4 * n + 5, 0);  
        lazy.assign(4 * n + 5, false);  
        build(1, 1, n);  
    }  
  
    // 原有函数保持不变  
    void push_up(int p) {  
        sumv[p] = sumv[p<<1] + sumv[p<<1|1];  
    }  
  
    void flip(int p, int l, int r) {  
        sumv[p] = (r - l + 1) - sumv[p];  
        lazy[p] = !lazy[p];  
    }  
  
    void push_down(int p, int l, int r) {  
        if (!lazy[p]) return;  
        int m = (l + r) >> 1;  
        flip(p<<1, l, m);  
        flip(p<<1|1, m+1, r);  
        lazy[p] = false;  
    }  
  
    void build(int p, int l, int r) {  
        lazy[p] = false;  
        if (l == r) {  
            sumv[p] = (s[l-1] == '1');  
            return;  
        }  
        int m = (l + r) >> 1;  
        build(p<<1, l, m);  
        build(p<<1|1, m+1, r);  
        push_up(p);  
    }  
  
    void update(int p, int l, int r, int ql, int qr) {  
        if (ql <= l && r <= qr) {  
            flip(p, l, r);  
            return;  
        }  
        push_down(p, l, r);  
        int m = (l + r) >> 1;  
        if (ql <= m) update(p<<1, l, m, ql, qr);  
        if (qr >  m) update(p<<1|1, m+1, r, ql, qr);  
        push_up(p);  
    }  
  
    int query(int p, int l, int r, int ql, int qr) {  
        if (ql <= l && r <= qr) return sumv[p];  
        push_down(p, l, r);  
        int m = (l + r) >> 1;  
        int res = 0;  
        if (ql <= m) res += query(p<<1, l, m, ql, qr);  
        if (qr >  m) res += query(p<<1|1, m+1, r, ql, qr);  
        return res;  
    }  
  
    // 外部调用接口  
    void range_flip(int l, int r) {  
        int n = s.length();  
        update(1, 1, n, l, r);  
    }  
  
    int range_ones(int l, int r) {  
        int n = s.length();  
        return query(1, 1, n, l, r);  
    }  
};  
  
void solve() {  
    int n, q;  
    cin >> n >> q;  
  
    string s;  
    cin >> s;  
  
    // 创建并使用线段树  
    SegmentTree seg;  
    seg.init(s);  
  
    while (q--) {  
        int op, l, r;  
        cin >> op >> l >> r;  
  
        if (op == 1) {  
            seg.range_flip(l, r);  
        } else {  
            cout << seg.range_ones(l, r) << "\n";  
        }  
    }  
}
```
