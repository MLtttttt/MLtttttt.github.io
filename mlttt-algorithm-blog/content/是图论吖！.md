
# 拓扑排序

## 描述
给定一张包含 n 个顶点、m 条边的**有向图**，请输出其**拓扑序**（Topological Order）。若存在多个合法序列，可输出任意一种；若图中存在有向环，无法拓扑排序，则输出 **−1。

```cpp 
void solve() {  
    int n, m;  
    cin >> n >> m;  
    vector<vector<int> > l(n + 1);  
    vector<int> d(n + 1);  
    for (int i = 0; i < m; i++) {  
        int a, b;  
        cin >> a >> b;  
        l[a].push_back(b);  
        d[b]++;  
    }  
    
    
//第一行输入两整数 n,m (1≦n,m≦2×1e5)n,m (1≦n,m≦2×1e5) —— 顶点数与边数。
//接下来 m行，每行输入两个整数 ui,vi (1≦ui,vi≦n)ui​,vi​ (1≦ui​,vi​≦n)，表示一条从 ui指向 vi​ 的有向边。 
    queue<int> q;  
    vector<int> ans;  
    for (int i = 1; i <= n; i++) {  
        if (d[i] == 0)q.push(i);  
    }  
    while (!q.empty()) {  
        int tp = q.front();  
        q.pop();  
        ans.push_back(tp);  
  
        for (auto &x: l[tp]) {  
            d[x]--;  
            if (d[x] == 0) {  
                q.push(x);  
            }  
        }  
    }  
  
    if (ans.size() == n) {  
        for (int i = 0; i < n; ++i)  
            cout << ans[i] << (i == n - 1 ? "" : " ");  
    } else cout << -1 << '\n';  
}  
```

## 二分图染色
### dfs

```cpp
  
bool dfs(int u, int c, vector<vector<pair<int, int>>>& gra, vector<int>& col) {  
    col[u] = c;  
    for (auto [v, tp]: gra[u]) {  
        if (col[v] == -1) {  
            int nc = (tp == 1) ? c : (1 - c);  // tp=1:同色，tp=0:异色  
            if (!dfs(v, nc, gra, col)) return false;  
        } else {  
            if (tp == 1 && col[v] != c) return false;  // 要求同色但不同  
            if (tp == 0 && col[v] == c) return false;  // 要求异色但相同  
        }  
    }  
    return true;  
}  
  
void solve() {  
    int n;  
    cin >> n;  
    vector<array<int, 3>> ed(n);  
    vector<int> num;  
  
    for (int i = 0; i < n; i++) {  
        int a, b, c;  
        cin >> a >> b >> c;  
        ed[i] = {a, b, c};  
        num.push_back(a);  
        num.push_back(b);  
    }  
  
    // 离散化  
    sort(num.begin(), num.end());  
    num.erase(unique(num.begin(), num.end()), num.end());  
    int siz = num.size();  
  
    // 使用局部变量而不是全局变量  
    vector<vector<pair<int, int>>> gra(siz);  
    vector<int> col(siz, -1);  
  
    auto id = [&](int x) {  
        return lower_bound(num.begin(), num.end(), x) - num.begin();  
    };  
  
    // 构建图  
    for (auto [a, b, c]: ed) {  
        int x = id(a), y = id(b);  
        if (c == 1) {  // 朋友关系，要求同色  
            gra[x].push_back({y, 1});  
            gra[y].push_back({x, 1});  
        } else {       // 敌人关系，要求异色  
            gra[x].push_back({y, 0});  
            gra[y].push_back({x, 0});  
        }  
    }  
  
    // DFS染色  
    for (int i = 0; i < siz; i++) {  
        if (col[i] == -1) {  
            if (!dfs(i, 1, gra, col)) {  // 从颜色1开始  
                cout << "NO" << '\n';  
                return;  
            }  
        }  
    }  
    cout << "YES" << '\n';  
}
```

```cpp

void solve() {  
    int n;  
    cin >> n;  
    vector<array<int, 3> > ed(n);  
    vector<int> num;  
    for (int i = 0; i < n; i++) {  
        int a, b, c;  
        cin >> a >> b >> c;  
        ed[i] = {a, b, c};  
        num.push_back(a);  
        num.push_back(b);  
    }  
  
    sort(num.begin(), num.end());  
    num.erase(unique(num.begin(), num.end()), num.end());  
    int siz = num.size();  
    auto id = [&](int x) {  
        return lower_bound(num.begin(), num.end(), x) - num.begin();  
    };  
  
    vector<vector<pair<int, int> > > gra(siz);  
    //  1朋友（同色），0 敌人（异色）  
    for (auto [a, b, c]: ed) {  
        int x = id(a), y = id(b);  
        if (c == 1) {  
            gra[x].push_back({y, 1});  
            gra[y].push_back({x, 1});  
        } else {  
            gra[x].push_back({y, 0});  
            gra[y].push_back({x, 0});  
        }  
    }  
  
    vector<int> col(siz, -1); // -1:未染色, 0:颜色0, 1:颜色1  
    for (int i = 0; i < siz; i++) {  
        if (col[i] == -1) {  
            queue<int> q;  
            q.push(i);  
            col[i] = 1;  
            while (!q.empty()) {  
                int u = q.front();  
                q.pop();  
                for (auto [v, tp]: gra[u]) {  
                    if (col[v] == -1) {  
                        if (tp == 1) col[v] = col[u]; //友同  
                        else col[v] = 1 - col[u]; //敌异  
                        q.push(v);  
                    } else {  
                        if (tp == 1) {  
                            if (col[v] != col[u]) {  
                                cout << "NO" << '\n';  
                                return;  
                            }  
                        } else {  
                            if (col[v] == col[u]) {  
                                cout << "NO" << '\n';  
                                return;  
                            }  
                        }  
                    }  
                }  
            }  
        }  
    }  
    cout << "YES" << '\n';  
}
```

## 离最近根节点的距离最大的节点

```cpp

void solve() {  
    int n;  
    cin >> n;  
    vector<vector<int> > g(n + 1); // 邻接表，下标从1开始  
    vector<int> d(n + 1, -1);      // 每个节点到最近叶子的距离，-1表示未访问  
    vector<int> fa(n + 1);         // 每个节点所属的最近叶子编号  
    queue<int> q;                  // BFS队列  
  
    // 读入n-1条边，构建邻接表  
    for (int i = 0; i < n - 1; i++) {  
        int u, v;  
        cin >> u >> v;  
        g[u].push_back(v);  
        g[v].push_back(u);  
    }  
  
    // 初始化：所有叶子节点（度数为1）入队，距离为0，所属叶子为自身  
    for (int i = 1; i <= n; ++i) {  
        if (g[i].size() == 1) {  
            d[i] = 0;  
            fa[i] = i;  
            q.push(i);  
        }  
    }  
  
    // 多源BFS，计算每个节点到最近叶子的距离及其归属  
    while (!q.empty()) {  
        int u = q.front();  
        q.pop();  
        for (int v : g[u]) {  
            if (d[v] == -1) {          // 如果v未访问过  
                d[v] = d[u] + 1;        // 距离加1  
                fa[v] = fa[u];           // 继承最近叶子的编号  
                q.push(v);  
            }  
        }  
    }//疑惑解决：检验了下，发现最接近的永远最早继承  
  
    // 找出所有节点中到最近叶子距离的最大值  
    int maxd = -1;  
    for (int i = 1; i <= n; ++i) maxd = max(maxd, d[i]);  
  
    // 收集所有达到最大距离的节点（注意：题目要求的是叶子节点，但此处是所有节点）  
    vector<int> ans;  
    for (int i = 1; i <= n; ++i) {  
        if (d[i] == maxd) ans.push_back(i);  
    }    
    // 输出结果：第一行个数，第二行编号（空格分隔）  
    cout << ans.size() << '\n';  
    for (int i = 0; i < ans.size(); i++) cout << ans[i] << ' ';  
    cout << '\n';  
}
```

## 根叶到其他根叶子的距离的最大值
```cpp
void solve() {  
    int n;  
    cin >> n;  
    vector<vector<int> > g(n + 1); 
    vector<int> d(n + 1, -1);    
    vector<int> fa(n + 1);       
    queue<int> q;                  
  
    for (int i = 0; i < n - 1; i++) {  
        int u, v;  
        cin >> u >> v;  
        g[u].push_back(v);  
        g[v].push_back(u);  
    }  
  
    for (int i = 1; i <= n; ++i) {  
        if (g[i].size() == 1) {  
            d[i] = 0;  
            fa[i] = i;  
            q.push(i);  
        }  
    }  
   
    while (!q.empty()) {  
        int u = q.front();  
        q.pop();  
        for (int v : g[u]) {  
            if (d[v] == -1) {          
                d[v] = d[u] + 1;      
                fa[v] = fa[u];         
                q.push(v);  
            }  
        }  
    }
      // 找出所有根叶中，到最近其他根叶子的距离的最大值  
        // leaf_dist[i]: 叶子i到最近的其他叶子的距离（初始化为无穷大）  
        vector<int> leaf_dist(n + 1, inf);  
        // 遍历每条边，如果两端点归属不同叶子，则计算这两个叶子之间的距离  
        // 并用该距离更新两个叶子的候选值（取较小者）  
        for (int u = 1; u <= n; ++u) {  
            for (int v : g[u]) {  
                if (u < v && fa[u] != fa[v]) {   // 避免重复边，且两端属于不同叶子  
                    int dd = d[u] + d[v] + 1;    // 两叶子间的路径长度  
                    leaf_dist[fa[u]] = min(leaf_dist[fa[u]], dd);  
                    leaf_dist[fa[v]] = min(leaf_dist[fa[v]], dd);  
                }  
            }  
        }  
        int maxd = -1;  
        for (int i = 1; i <= n; ++i) {  
            if (g[i].size() == 1)  maxd = max(maxd, leaf_dist[i]);  
        }  
  
        vector<int> ans;  
        for (int i = 1; i <= n; ++i) {  
            if (g[i].size() == 1 && leaf_dist[i] == maxd) {  
                ans.push_back(i);  
            }  
        }  
  
    // 输出结果：第一行个数，第二行编号（空格分隔）  
    cout << ans.size() << '\n';  
    for (int i = 0; i < ans.size(); i++) cout << ans[i] << ' ';  
    cout << '\n';  
}  

```


## 数字变换的连通块乘积
```cpp
void solve() {
    int x, n;
    cin >> x >> n;
    vector<bool> vis(10, false);
    vector<vector<int> > l(10, vector<int>(10, 0));
    int t = n;
    while (t--) {
        int a, b;
        cin >> a >> b;
        l[a][b] = 1;
    }

    i64 ans = 1;
    while (x){
        vis.assign(n , false);
        queue<int> q;
        int i = x%10;
        x /= 10;
        q.push(i);
        i64 cnt = 0;
        while (q.size()) {
            i = q.front();
            q.pop();
          if (vis[i])continue;
            vis[i]=true;
            cnt++;
            for (int j = 0; j <= 9; j++) {
                if (j == i)continue;
                if (l[i][j]&&!vis[j]) q.push(j);
            }
        }
        ans = ans * cnt % mod;
    }
    cout << ans << '\n';
}
```
