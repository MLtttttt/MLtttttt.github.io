## 路径问题

题目：链接：[https://ac.nowcoder.com/acm/contest/119696/C](https://ac.nowcoder.com/acm/contest/119696/C)  来源：牛客网  
  
已知迷宫由一个 n 行 m 列的地图矩阵组成，若小 A 当前位于 (x,y)，则小 A 可以在不离开迷宫的前提下选择移动到上 (x - 1, y) ，下 (x+1,y))、左 (x,y−1)、右 (x,y+1) 四格。迷宫的格子可能为碎冰或者平整冰面：
当小 A 离开平整冰面后，原先所站位置上的平整冰面就会变为碎冰；
当碎冰被小A站上后，小 A 会掉落到战斗区，和一个宝可梦训练师战斗爽。
他只想在他想掉落的方格掉落，以便能让他战胜对手。
小 A 的起始位置 (sx,sy) 和目标掉落位置 (ex,ey)
请你写一个程序，帮助小 A 判断是否能在他想掉落的方格掉落。
保证起点为平整冰面，==终点为碎冰。==
```cpp
 
vector<vector<int> > a;  
  
int n, m, sx, sy, ex, ey;  
int dx[4] = {-1, 1, 0, 0};  
int dy[4] = {0, 0, -1, 1};  
bool ok = false;  
  
void dfs(int x, int y) {  
    a[x][y] = 0;  
    if (ok) return;  
  
    for (int k = 0; k < 4; k++) {  
        int X = x + dx[k];  
        int Y = y + dy[k];  
        if (X == ex && Y == ey) {  
            ok = true;  
            return;  
        }  //根据题意放在外面还是里面
        
        if (X > 0 && X <= n && Y > 0 && Y <= m && a[X][Y] != 0) dfs(X, Y);  
    }  
}  
  
void solve() {  
    cin >> n >> m;  
    a.resize(n + 2, vector<int>(m + 2));  
  
    for (int i = 1; i <= n; i++) {  
        string s;  
        cin >> s;  
        for (int j = 1; j <= m; j++) {  
            if (s[j - 1] == '*') a[i][j] = 0;  
            else a[i][j] = 1;  
        }  //地图输入
    }  
    cin >> sx >> sy >> ex >> ey;  
    ok = false;  
    dfs(sx, sy);  
  
    if (ok) cout << "yes" << '\n';  
    else cout << "no" << '\n';  
}  
  
int main() {  
    cin.tie(nullptr), ios::sync_with_stdio(false);  
  
    int t = 1;  
    while (t--) {  
        solve();  
    }  
    return 0;  
}
```

```cpp
void solve() {
    int n, m, q;
    cin >> n >> m >> q;
    
    // 修正：使用正确的维度 n+2 x m+2
    vector<vector<i64>> a(n + 2, vector<i64>(m + 2, 0));
    
    // 读取初始矩阵
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            cin >> a[i][j];
        }
    }
    
    // 构建差分数组
    vector<vector<i64>> d(n + 2, vector<i64>(m + 2, 0));
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            d[i][j] = a[i][j] - a[i-1][j] - a[i][j-1] + a[i-1][j-1];
        }
    }

    for (int i = 0; i < q; i++) {
        int x1, y1, x2, y2, k;
        cin >> x1 >> y1 >> x2 >> y2 >> k;
        
        d[x1][y1] += k;
        d[x1][y2 + 1] -= k;
        d[x2 + 1][y1] -= k;
        d[x2 + 1][y2 + 1] += k;
    }
    // 通过差分数组恢复最终矩阵
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            a[i][j] = d[i][j] + a[i-1][j] + a[i][j-1] - a[i-1][j-1];
        }
    }
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            cout << a[i][j] << " \n"[j == m];
        }
    }
}
```

## 组合生成问题
```cpp
#include <bits/stdc++.h>  
using namespace std;  

vector<bool> vis;  
vector<int> a;  
int n, m;  
  
void dfs(int id, int s) {  
    if (id > m) {  
        for (int i = 1; i <= m; i++) cout << a[i] << ' ';  
        cout << '\n';  
        return;  
    }  
    for (int i = s; i <= n; i++) {  
        if (!vis[i]) {  
            vis[i] = true;  
            a[id] = i;  
            dfs(id + 1, i + 1);  
            vis[i] = false;  
        }  
    }  
}  
  
void solve() {  
    cin >> n >> m;  
    a.resize(m + 1);  
    vis.assign(n + 1, false);  
    dfs(1, 1);  
}  
```


## 贪心排序求最小最大值（国王游戏）

```cpp
  
  
vector<int> mul(const vector<int> &a, int b) {  
    vector<int> res;  
    int carry = 0;  
    for (int x: a) {  
        carry += x * b;  
        res.push_back(carry % 10);  
        carry /= 10;  
    }    while (carry) {  
        res.push_back(carry % 10);  
        carry /= 10;  
    }    return res;  
}  
  
vector<int> div(const vector<int> &a, int b) {  
    vector<int> res;  
    int la = 0;  
    for (int i = a.size() - 1; i >= 0; i--) {  
        la = la * 10 + a[i];  
        res.push_back(la / b);  
        la %= b;  
    }    reverse(res.begin(), res.end());  
    while (res.size() > 1 && res.back() == 0) res.pop_back();  
    return res;  
}  
  
struct p {  
    i64 a, b, cj;  
};  
  
void solve() {  
    int n;  
    cin >> n;  
    i64 ka, kb;  
    cin >> ka >> kb;  
    vector<p> m(n);  
    for (auto &[x, y, z]: m) {  
        cin >> x >> y;  
        z = x * y;  
    }    sort(m.begin(), m.end(), [](const p &aa, const p &bb) {  
        return aa.cj < bb.cj;  
    });  
    vector<int> fz;  
    int t = ka;  
    do {  
        fz.push_back(t % 10);  
        t /= 10;  
    } while (t);  
  
    vector<int> ans;  
    ans.push_back(0);  
  
    for (auto &[x, y, z]: m) {

                    ok = (cur[i] > ans[i]);  
                    break;  
                }            }        }        if (ok) ans = cur;  
        fz = mul(fz, x);  
    }  
    for (int i = ans.size() - 1; i >= 0; i--) cout << ans[i];  
    cout << '\n';  
}
```

##  满二叉树短路径个数
```cpp
i64 qpow(i64 a, i64 b) {  
    i64 res = 1;  
    while (b) {  
        if (b & 1) res = res * a % mod;  
        a = a * a % mod;  
        b >>= 1;  
    }  
    return res;  
}  
  
void solve() {  
    int n;  
    cin >> n;  
    if (n == 1) {  
        cout << 0 << '\n';  
        return;  
    }  
    if (n == 2) {  
        cout << 1 << '\n';  
        return;  
    }  
  
    i64 sum = 0;  
    sum = (qpow(2, n - 2) - 1) * 6 % mod;  
    cout << sum + 1 << '\n';  
}
```
