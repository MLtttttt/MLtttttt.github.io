#include <stdint.h>  
#include <bits/stdc++.h>  
  
using namespace std;  
  
using i128 = __int128;  
using f80 = long double;  
using u64 = unsigned long long;  
using i64 = long long;  
using u32 = unsigned;  
  
constexpr int inf = 1e9;  
constexpr i64 infl = 1e18;  
  
constexpr f80 eps = 1e-8; // 根据题目精度要求进行修改  
const f80 pi = acos(-1.0L); // 3.1415926....  
  
// 取符号函数  
int sgn(auto x) {  
    // 精度范围内的近似相等  
    if (abs(x) < eps) return 0;  
    return x > 0 ? 1 : -1;  
}  
  
template<typename T>  
struct pos {  
    T x, y;  
  
    template<typename U>  
    operator pos<U>() const {  
        static_assert(is_convertible_v<T, U>, "T must be convertible to U");  
        return {static_cast<U>(x), static_cast<U>(y)};  
    }  
  
    pos operator+(const pos &b) const { return {x + b.x, y + b.y}; }  
  
    pos operator-(const pos &b) const { return {x - b.x, y - b.y}; }  
  
    T operator^(const pos &b) const { return x * b.y - y * b.x; }  
  
    T operator*(const pos &b) const { return x * b.x + y * b.y; }  
  
    pos operator*(const T &b) const { return {x * b, y * b}; }  
  
    pos operator/(const T &b) const { return {x / b, y / b}; }  
  
    bool operator==(const pos &b) const { return !sgn(x - b.x) && !sgn(y - b.y); }  
  
    [[nodiscard]] T abs2() const { return x * x + y * y; }  
};  
  
using Pi = pos<i64>;  
using Pf = pos<f80>;  
  
// 获得点 a 到点 b 的距离  
f80 dist(auto a, auto b) {  
    return sqrt((a - b).abs2());  
}  
  
// 获得点 a 的单位化向量  
auto norm(auto a) {  
    return a / hypot(a.x, a.y);  
}  
  
// 判断点 p 在向量 ab 的哪一侧  
int side(auto p, auto a, auto b) {  
    return sgn((a - p) ^ (b - p));  
}  
  
// 判断点 p 是否在线段 ab 上  
bool ons(auto p, auto a, auto b) {  
    return side(p, a, b) == 0 && sgn((a - p) * (b - p)) <= 0;  
}  
  
// 判断线段 ab 与线段 cd 是否存在交点  
bool iss(auto a, auto b, auto c, auto d) {  
    if (ons(a, c, d) || ons(b, c, d) || ons(c, a, b) || ons(d, a, b)) return true;  
    if (side(a, b, c) * side(a, b, d) >= 0) return false;  
    if (side(c, d, a) * side(c, d, b) >= 0) return false;  
    return true;  
}  
  
// 判断直线 ab 与线段 cd 是否存在交点  
bool ils(auto a, auto b, auto c, auto d) {  
    return side(a, b, c) * side(a, b, d) <= 0;  
}  
  
// 判断直线 ab 与直线 cd 是否平行  
bool para(auto a, auto b, auto c, auto d) {  
    return sgn(a - b ^ c - d) == 0;  
}  
  
// 获得直线 ab 与直线 cd 的交点  
Pf ill(Pf a, Pf b, Pf c, Pf d) {  
    auto u = b - a, v = d - c;  
    auto t = (a - c ^ v) / (v ^ u);  
    return a + u * t;  
}  
  
// 获取点 p 在直线 uv 上的垂足  
Pf foot(Pf p, Pf u, Pf v) {  
    auto dir = norm(u - v);  
    return u + dir * (dir * (p - u));  
}  
  
// 获取向量 a 逆时针旋转角度 r 后的向量， r ∈ [0, PI]Pf rot(Pf a, f80 r) {  
    Pf b = {sin(r), cos(r)};  
    return {a ^ b, a * b};  
}  
  
// 获取三角形 ABC 的面积  
f80 tria(auto a, auto b, auto c) {  
    return abs(b - a ^ c - a) / 2;  
}  
  
// 获取三角形 ABC 的面积  
i64 tria2(Pi a, Pi b, Pi c) {  
    return abs(b - a ^ c - a);  
}  
  
// 获取闭合多边形 p 的面积  
f80 polya(auto &p) {  
    f80 res = 0;  
    for (int i = 0; i < p.size(); i++) res += p[i] ^ p[(i + 1) % p.size()];  
    return abs(res) / 2;  
}

}

    cout << fixed << setprecision(18);  
  
    int t = 1;  
    // cin >> t;

        solve();  
    }  
  
    return 0;  
}

void example1_point_in_triangle() {
    // 问题：判断点P是否在三角形ABC内部
    Pi A{0, 0}, B{4, 0}, C{2, 3};
    Pi P{2, 1};
    
    // 方法：利用叉积判断P是否在三条边的同一侧
    int side1 = side(P, A, B);  // P相对于AB的方位
    int side2 = side(P, B, C);  // P相对于BC的方位
    int side3 = side(P, C, A);  // P相对于CA的方位
    
    // 如果都在同一侧（或边上），则在三角形内
    bool inside = (side1 >= 0 && side2 >= 0 && side3 >= 0) ||
                  (side1 <= 0 && side2 <= 0 && side3 <= 0);
    
    cout << "点P" << (inside ? "在" : "不在") << "三角形ABC内部" << endl;
    
    // 计算三角形面积验证
    f80 area = tria(Pf(A), Pf(B), Pf(C));
    cout << "三角形ABC面积: " << area << endl;
}

void example2_intersection_and_foot() {
    // 问题：求直线AB与线段CD的交点，以及点E到直线AB的垂足
    Pf A{0, 0}, B{4, 2};      // 直线AB
    Pf C{1, 3}, D{3, 1};      // 线段CD
    Pf E{2, 0};               // 点E
    
    // 1. 判断直线AB与线段CD是否相交
    if (ils(A, B, C, D)) {
        cout << "直线AB与线段CD相交" << endl;
        
        // 2. 计算交点
        Pf intersection = ill(A, B, C, D);
        cout << "交点坐标: (" << intersection.x << ", " << intersection.y << ")" << endl;
        
        // 验证点是否在线段上
        if (ons(intersection, C, D)) {
            cout << "交点在线段CD上" << endl;
        }
    }
    
    // 3. 计算点E到直线AB的垂足
    Pf foot_point = foot(E, A, B);
    cout << "点E到直线AB的垂足: (" << foot_point.x << ", " << foot_point.y << ")" << endl;
    
    // 4. 计算距离
    f80 dist_to_line = dist(E, foot_point);
    f80 dist_AB = dist(A, B);
    cout << "点E到直线AB的距离: " << dist_to_line << endl;
    cout << "AB长度: " << dist_AB << endl;
}

```cpp
void example3_polygon_area_and_point() {
    // 问题：计算多边形面积并判断点是否在多边形内（射线法）
    vector<Pf> polygon = {
        {0, 0}, {4, 0}, {4, 3}, {2, 5}, {0, 3}
    };
    Pf P{2, 2};
    Pf Q{5, 5};  // 外部点
    
    // 1. 计算多边形面积
    f80 area = polya(polygon);
    cout << "多边形面积: " << area << endl;
    
    // 2. 简单射线法判断点是否在多边形内
    auto ray_casting = [&](Pf point) -> bool {
        int count = 0;
        Pf ray_end{inf, point.y};  // 向右的水平射线
        
        for (int i = 0; i < polygon.size(); i++) {
            Pf p1 = polygon[i];
            Pf p2 = polygon[(i + 1) % polygon.size()];
            
            // 如果点在顶点上
            if (point == p1 || point == p2) return true;
            
            // 如果点在水平边线上
            if (sgn(p1.y - point.y) == 0 && sgn(p2.y - point.y) == 0 &&
                sgn(min(p1.x, p2.x) - point.x) <= 0 &&
                sgn(max(p1.x, p2.x) - point.x) >= 0) {
                return true;
            }
            
            // 常规相交判断
            if (sgn((p1.y - point.y) * (p2.y - point.y)) < 0) {
                Pf intersection = ill(point, ray_end, p1, p2);
                if (sgn(intersection.x - point.x) > 0) {
                    count++;
                }
            }
        }
        return count % 2 == 1;
    };
    
    cout << "点P" << (ray_casting(P) ? "在" : "不在") << "多边形内" << endl;
    cout << "点Q" << (ray_casting(Q) ? "在" : "不在") << "多边形内" << endl;
}
```

void example5_circumcenter() {
    // 问题：已知三角形ABC三个顶点，求外接圆圆心
    Pf A{0, 0}, B{4, 0}, C{2, 3};
    
    // 方法：做两边垂直平分线的交点
    Pf AB_mid = (A + B) / 2.0;  // AB中点
    Pf BC_mid = (B + C) / 2.0;  // BC中点
    
    // AB的垂直方向向量
    Pf AB_dir = B - A;
    Pf AB_perp = rot(AB_dir, pi/2);  // 旋转90度得到垂直向量
    
    // BC的垂直方向向量
    Pf BC_dir = C - B;
    Pf BC_perp = rot(BC_dir, pi/2);
    
    // 做垂直平分线求交点
    Pf center = ill(AB_mid, AB_mid + AB_perp, BC_mid, BC_mid + BC_perp);
    
    // 验证：三个顶点到圆心的距离相等
    f80 r1 = dist(center, A);
    f80 r2 = dist(center, B);
    f80 r3 = dist(center, C);
    
    cout << "三角形ABC外接圆圆心: (" << center.x << ", " << center.y << ")" << endl;
    cout << "半径: " << r1 << endl;
    cout << "验证三个半径是否相等:" << endl;
    cout << "  A到圆心距离: " << r1 << endl;
    cout << "  B到圆心距离: " << r2 << endl;
    cout << "  C到圆心距离: " << r3 << endl;
    cout << "  最大误差: " << max({abs(r1-r2), abs(r2-r3), abs(r3-r1)}) << endl;
}

## **关键知识点笔记**

### **1. 叉积的应用**
```cpp
(a ^ b) > 0  // b在a的逆时针方向
(a ^ b) < 0  // b在a的顺时针方向
(a ^ b) = 0  // a与b共线
```

### **2. 点积的应用**
```cpp
(a * b) > 0  // 夹角小于90度
(a * b) < 0  // 夹角大于90度
(a * b) = 0  // 垂直
```

### **3. 常用技巧**
- **判断点在线段上**：先判断共线，再判断点积≤0
- **面积计算**：用叉积的绝对值/2
- **精度处理**：使用`eps`避免浮点误差

### **4. 性能优化**
- 整数坐标用`Pi`，避免浮点误差
- 需要高精度时用`Pf`
- 比较时用`sgn()`代替直接比较
