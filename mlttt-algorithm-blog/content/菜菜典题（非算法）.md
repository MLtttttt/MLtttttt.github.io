给定一个正整数 n，按照以下规则输出字符三角形：

1. 整体形状为跟配图类似的等腰三角形（当 n>1 时需在行首添加适量空格）
2. 共输出 n 行，第 k 行（1≤k≤n）包含2k−1 个字母
3. 每行的字母序列左右两侧关于中心字母对称，具体要求为：
- 左侧：从A 递增至当前行中心字母（如 A→B→C）
- 中心：当前行最高字母（如第 3 行中心为 C）
- 右侧：从中心字母递减回C→B→A）
  ![[eeb653af-2221-40a4-a54b-0ee116ecf339 1.png]]

```cpp
void solve() {
    int n;
    cin >> n;
    for (int i = 0; i < n; i++) {
        for (int b = 0; b < n - 1 - i; b++)cout << " ";
        string s;
        for (char ch = 'A'; ch <= 'A' + i; ch++)s += ch;
        for (char ch = 'A' + i - 1; ch >= 'A'; ch--) s += ch;
        cout << s;
        cout << '\n';
    }
}
```

##  二进制拆解移动

链接：[https://ac.nowcoder.com/acm/contest/119118/H](https://ac.nowcoder.com/acm/contest/119118/H)  
来源：牛客网  

鸡哥养了 n 只鸡，现在要把它们装进 k个鸡圈里.
鸡圈特性
：第一个鸡圈能装 1 只鸡，第二个能装 2 只鸡，第三个能装 4 只鸡 …… 
每个鸡圈的容量都是前一个的两倍。
鸡哥规定：**每个鸡圈要么装满，要么空着**。  
判断能否利用这 k 个鸡圈恰好装下 n只鸡？如果能，输出 `YES`；如果不能，输出 `NO`。

```cpp
void solve() {
    i64 n, k;
    cin >> n >> k;
    i64 sum = (1LL << k) - 1;
    //1LL 表示长整型的 1（避免整数溢出）
     //左移 k 位，等价于  2^k （比如 k=3 时，1 << 3 是  2^3 = 8 ）。

    if (n <= sum)cout << "YES" << '\n';
    else cout << "NO" << '\n';
}
```

# 异或 按位或 按位与
题目
通过xor 希望 a 的值恰好变为 b 。
思路（找中间态）
a ^ x1 == temp
b^ x2 == temp
a^x1 == b^ x2
```cpp
u64 maxh(u64 x) {
    if (x == 0) return 0;
    u64 hh = 1;
    while (hh <= x / 2) {
        hh <<= 1; // 等价于 hh *= 2，左移操作效率更高
    }
    return hh;
}
void solve() {
    u64 a, b;
    cin >> a >> b;
    if (a == b) { // 情况1：直接相等
        cout << "0\n";
        return;
    }
    u64 ha = maxh(a); // a的最高位
    u64 hb = maxh(b); // b的最高位
    if (hb > ha) { // 情况2：b的最高位更大，无法操作
        cout << "-1\n";
        return;  
    } 
    else if (hb < ha) { // 情况3：b的最高位更小，两次操作
        u64 x1 = a ^ (ha | hb); // 第一次操作：a → ha|hb（中间态）
        u64 x2 = (ha | hb) ^ b; // 第二次操作：中间态 → b
        cout << "2\n";
        cout << x1 << " " << x2 << "\n";
    }
     else { // 情况4：最高位相等，优先一次操作
        u64 x = a ^ b;
        if (x <= a) { // 一次操作可行
            cout << "1\n";
            cout << x << "\n";
        } else { // 一次操作不可行，退而两次操作
            u64 x1 = a ^ (ha | hb);
            u64 x2 = (ha | hb) ^ b;
            cout << "2\n";
            cout << x1 << " " << x2 << "\n";
        }
    }
}
```


## 二分找第y小的数

题目要求：有一个包含N个不同整数的列表A。有Q个查询，每个查询给出两个整数Xj和Yj。对于每个查询，需要找到不在列表A中，且大于等于Xj的整数中，第Yj小的值。

```cpp

void solve() {  
    int n, q;  
    cin >> n >> q;  
    vector<i64> a(n);  
    for (int i = 0; i < n; i++) cin >> a[i];  
    sort(a.begin(), a.end());  
  
    while (q--) {  
        i64 x, y;  
        cin >> x >> y;  
        i64 l = x, r = x + y + n, mid;  
        auto itl = lower_bound(a.begin(), a.end(), x);  
        while (l <= r) {  
            mid = l + (r - l) / 2;  
            auto itr = upper_bound(a.begin(), a.end(), mid);  
            i64 ban = itr - itl;  
            i64 g = (mid - x + 1) - ban;  
            if (g >= y) r = mid - 1;  
            else l = mid + 1;  
        }  
        cout << l << "\n";  
    }  
}
```

## DP合并计数：右向左 + 模10

```cpp

void solve() {  
    int n;  
    cin >> n;  
    vector<int> a(n);  
  
    for (int i = 0; i < n; i++) {  
        cin >> a[i];  
        if (n == 1 && a[0] >= 10) {  
            for (int i = 0; i < 10; i++) cout << 0 << ' ';  
            return;  
        }  
        a[i] %= 10;  
    }  
  
    vector<i64> dp(10, 0);  
  
    // 初始化：最后一个数字，方案数为1  
    dp[a[n - 1]] = 1;  
  
    // 从右向左处理数组（因为每次操作最后两个数）  
    for (int i = n - 2; i >= 0; i--) {  
        // 新状态数组  
        vector<i64> np(10, 0);  
        // 遍历所有可能的结果（0-9）  
        for (int d = 0; d < 10; ++d) {  
            // 优化：如果dp[d]为0，跳过  
            if (dp[d] == 0) continue;  
  
            // 两种操作结果  
            int ad = (a[i] + d) % 10;  // 加法取个位  
            int mu = (a[i] * d) % 10;  // 乘法取个位  
  
            // 状态转移：方案数累加  
            np[ad] = (np[ad] + dp[d]) % mod;  
            np[mu] = (np[mu] + dp[d]) % mod;  
        }  
          
        // 更新dp数组（三种方式等价）  
        dp = move(np);  // 移动语义，高效  
        // swap(dp, np); // 交换方式，同样高效  
        // dp = np;      // 复制方式，效率较低  
    }  
  
    for (int i = 0; i < 10; ++i)  cout << dp[i] << ' ';  
}
```

## 小根堆合并代价

```cpp  
void solve() {  
    i64 n;  
    cin >> n;  
    i64 sm = 0;  
    vector<i64> c(n + 1), w(n + 1);  
    priority_queue<pair<i64, i64>, vector<pair<i64, i64> >, greater<pair<i64, i64> > > pq;  
  
    for (int i = 1; i <= n; i++) {  
        cin >> c[i];  
        cin >> w[i];  
        pq.emplace(w[i], c[i]);  
    }  
  
    while (pq.size() > 1) {  
        auto [x1,y1] = pq.top();  
        pq.pop();  
        if (y1 >= 2) {  
            i64 tk = y1 / 2;  
            sm = (sm + (2 * x1 % mod) * tk % mod) % mod;  
            pq.emplace(x1 * 2, tk);  
            if (tk * 2 < y1)pq.emplace(x1, 1);  
        } else {  
            auto [x2,y2] = pq.top();  
            pq.pop();  
            i64 tp = x1 + x2, k = min(y1, y2);  
            if (k)pq.emplace(tp, k);  
            sm = (sm + tp % mod * k % mod) % mod;  
            if (y1 > k)pq.emplace(x1, y1 - k);  
            else if (y2 > k)pq.emplace(x2, y2 - k);  
        }  
    }  
    cout << sm % mod << '\n';  
}  

```

## 相邻字符互异的子序列计数

```cpp
void solve() {  
    string s;  
    cin >> s;  
    i64 sum = 0;  
    map<char, i64> cnt;  
    for (int i = 0; i < s.size(); i++) {  
        i64 la = cnt[s[i]];  
        i64 na = 1 + (sum - la + mod) % mod;  
        cnt[s[i]] = (la + na) % mod;  
        sum = (sum + na) % mod;  
    }  
  
    cout << sum % mod << '\n';  
}
```

## 1 多于 0 的子矩阵计数

```cpp

void solve() {  
    int n, m;  
    cin >> n >> m;  
    vector<vector<int>> a(n + 1, vector<int>(m + 1));  
    for (int i = 1; i <= n; i++)  
        for (int j = 1; j <= m; j++)cin >> a[i][j];  
    vector<vector<int>> pre(n + 1, vector<int>(m + 1));  
    for (int i = 1; i <= n; i++)  
        for (int j = 1; j <= m; j++)  
            pre[i][j] = pre[i-1][j] + pre[i][j-1] - pre[i-1][j-1] + a[i][j];  
  
    i64 ans = 0;  
    for (int x1 = 1; x1 <= n; x1++) {  
        for (int y1 = 1; y1 <= m; y1++) {  
            for (int x2 = x1; x2 <= n; x2++) {  
                for (int y2 = y1; y2 <= m; y2++) {  
                    int sum = pre[x2][y2] - pre[x1-1][y2] - pre[x2][y1-1] + pre[x1-1][y1-1];  
                    int area = (x2 - x1 + 1) * (y2 - y1 + 1);  
                    if (sum > area - sum)ans++;  
                }  
            }  
        }  
    }  
    cout << ans << '\n';  
}
```

## 多个子串的最小修改次数
```cpp
string s1 = "awdec";  
string s2 = "Fantasy_Blue";  
int n, k;  
string s;  
int l1 = s1.size(), l2 = s2.size();  
  
void solve() {  
    cin >> n >> k >> s;  
  
    if ((int) s.size() < 17) {  
        cout << "No\n";  
        return;  
    }  
    if (k >= 17) {  
        cout << "Yes\n";  
        return;  
    }  
  
    i64 na = -1, nb = -1;  
    vector<int> w1; // 存储达到最大匹配的起始位置  
    for (int i = 0; i <= n - l2; i++) {  
        i64 cnt = 0;  
        for (int j = 0; j < l2; j++) {  
            if (s[i + j] == s2[j]) cnt++;  
        }  
        if (cnt >= na) {  
            if (cnt > na) w1.clear();  
            na = cnt;  
            w1.push_back(i);  
        }  
    }  
  
    int l = w1[0] + l2;  
    int r = w1.back();  
    // 先搜索 [0, min(r-5, n-5)] 区间  
    for (int i = 0; i <= r - 5; i++) {  
        i64 cnt = 0;  
        for (int j = 0; j < l1; j++) {  
            if (s[i + j] == s1[j]) cnt++;  
        }  
        if (cnt > nb) nb = cnt;  
    }  
    // 再搜索 [l, n-5] 区间  
    for (int i = l; i <= n - 5; i++) {  
        i64 cnt = 0;  
        for (int j = 0; j < l1; j++) {  
            if (s[i + j] == s1[j]) cnt++;  
        }  
        if (cnt > nb) nb = cnt;  
    }  
  
    if ( na + nb + k < 17)cout << "No\n";  
    else cout << "Yes\n";  
}
```

## 三角形计数（等腰底平行）
```cpp

void solve() {  
    int n;  
    cin >> n;  
    map<int, vector<int>> ax, ay;  
    map<int, int> cx, cy;  
    map<pair<int, int>, int> vis;  
  
    for (int i = 0; i < n; i++) {  
        int x, y;  
        cin >> x >> y;  
        ax[y].push_back(x);  
        ay[x].push_back(y);  
        cx[x]++, cy[y]++;  
        vis[{x, y}] = 1;  
    }  
    i64 ans = 0;  
  
    for (auto &[y, xs]: ax) {  
        int m = xs.size();  
        for (int i = 0; i < m; i++) {  
            for (int j = i + 1; j < m; j++) {  
                int x1 = xs[i], x2 = xs[j];  
                if ((x1 + x2) % 2) continue;  
                int mid = (x1 + x2) / 2;  
                int tp = cx[mid];  
                if (vis.count({mid, y})) tp--;  
                ans += tp;  
            }  
        }  
    }  
  
    for (auto &[x, ys]: ay) {  
        int m = ys.size();  
        for (int i = 0; i < m; i++) {  
            for (int j = i + 1; j < m; j++) {  
                int y1 = ys[i], y2 = ys[j];  
                if ((y1 + y2) % 2) continue;  
                int mid = (y1 + y2) / 2;  
                int tp = cy[mid];  
                if (vis.count({x, mid})) tp--;  
                ans += tp;  
            }  
        }  
    }  
  
    cout << ans << '\n';  
}
```
