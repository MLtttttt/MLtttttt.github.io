# 二分法

## l r mid

```cpp
void solve() {
    int m, n;
    cin >> m >> n;
    vector<int> a(m), b(n); // 声明和读入可以压下行，欧克
    for (int i = 0; i < m; i++) cin >> a[i];
    sort(a.begin(), a.end());
    for (int i = 0; i < n; i++) cin >> b[i];

    i64 sum = 0;
    for (int val: b) {
        int cur = inf;
        int l = 0, r = m - 1, pos = -1;
        while (l <= r) {
            int mid = (l + r) / 2;
            if (a[mid] <= val) {
                pos = mid;
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }
        if (pos != -1) cur = min(cur, abs(a[pos] - val));
        if (pos + 1 < m) cur = min(cur, abs(a[pos + 1] - val));
        sum += cur;
    }
    cout << sum << '\n';
}
```

# 

 ** 无优化（朴素实现）：
 **O(n)
 ***** 每次查找可能需要遍历整个树的高度，在极端情况下（如树退化成链表），时间复杂度很高。
 		
** 仅路径压缩：	
**O(α(n))	
*****  路径压缩将查找路径上的所有节点直接指向根节点，大大降低了树的高度。	

** 仅按秩合并：
**O(log n)
***** 按秩合并（或按大小合并）确保在合并两个集合时，较小的树合并到较大的树中，从而控制树的高度增长。	

** 路径压缩 + 按秩合并：
**O(α(n))
***** 这是并查集的最优优化组合，将每次操作的时间复杂度降低到接近常数的 O(α(n))。	

###  路径压缩+按高度合并实现

```cpp

int fa[N], siz[N];
int n, m;//写vector

int find(int x){
	if(fa[x] == x) return x; // 已经是根节点 
	return fa[x] = find(fa[x]); // 继续找 
}

int main(){
	cin >> n >> m;
	for(int i = 1; i <= n; ++i) fa[i] = i, siz[i] = 1; // 初始化
	for(int i = 1; i <= m; ++i){
		int op, x, y;
		cin >> op >> x >> y;
		if(op == 1){ // 合并操作 
			x = find(x), y = find(y); // 查询各自的代表元素
			if(x == y) continue; // 如果已经在同一个集合，跳过
			if(siz[y] < siz[x]) swap(x, y); // 改变合并顺序
			if(siz[x] == siz[y]) siz[y] = siz[x]+1;
			 // 计算新的秩：如果两个集合的秩相等，那么秩加一
			// 否则秩不变

		}
		else{ // 查询操作的一种变形
```

```cpp
    else return fa[x] = find(fa[x]);

    for (int i = 1; i <= n; ++i) fa[i] = i, siz[i] = 1;
    for (int i = 1; i <= m; ++i) {
        int op, x, y, z;
        cin >> op;
    	if(op == 1){ // 合并操作 
			x = find(x), y = find(y); // 查询各自的代表元素
			if(x == y) continue; // 如果已经在同一个集合，跳过
			if(siz[y] < siz[x]) swap(x, y); // 改变合并顺序
			siz[y] += siz[x]; // 计算新的秩
			fa[x] = y; // 合并 
		}
		else if(op == 2){ // 查询操作的一种变形 
			x = find(x), y = find(y); // 查询各自的代表元素
			if(x == y) cout << "Y" << endl; // 是否在同一个集合

            cin >> z;
            z = find(z);
            cout << siz[z] << '\n';
        }
    }
    return ;
}
```

```cpp
#include <bits/stdc++.h>  
using namespace std;  
  
using f80 = long double;

using i128 = __int128;

constexpr int mxv = 1e6 + 1;  
constexpr i64 m = 1e9 + 5;  
  
int n, k;  
  
//用vector的话，加&，数据大也用  
void ask(vector<int> &a) {  
    cout << "? ";  
    for (int i = 0; i < n; i++)  
        cout << a[i] << " ";  
    cout.flush();  
} //需要返回值时候改int ask  
  
void tell(vector<int> &b) {  
    cout << "! ";  
    for (int i = 0; i < n; i++)  
        cout << b[i] << " ";  
    cout.flush();  
}  
  
void solve() {  
    cin >> n;  
    vector<int> a(n, 0);  
    int idx = 0, ans = 0, val = 1;  
    while (ans < n) {  
        vector<int> tmp(n, 0);  
        tmp[idx] = val;  
        ask(tmp);  
        cin >> k;  
        if (k) {  
            a[idx] = val;  
            ans++;  
            idx = 0;  
            val++;  
        } else idx++;  
    }  
    tell(a);  
}  
  
int main() {  
    ios::sync_with_stdio(false), cin.tie(nullptr);  
  
    int t;  
    cin >> t;  
    while (t--) {  
        solve();  
    }  
  
    return 0;  
}
```
